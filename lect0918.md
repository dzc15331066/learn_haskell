#Curry form
##mult :: Int -> Int -> Int
##mult x y = x * y
##(Bool -> Bool) -> Bool is not the same as Bool->(Bool->Bool)

#Uncurried form
##mult2 :: (Int, Int) -> Int
##mult2 (x,y) = x * y
